int sum(int maks){
	if(maks == 0){
		return maks;
	}
	else if( maks > 1 ){
		while( maks % 2==0 ){
			maks--;
		}
		if ( maks != 1 ){
			cout << maks << " + ";	
		}
		else {
			cout << maks << " = ";
		}
		return maks + sum(maks-1);
	}
}
